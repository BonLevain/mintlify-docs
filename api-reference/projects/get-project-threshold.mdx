---
title: "Get Project Billing Threshold"
api: "GET /v2/project/{project_id}/threshold"
description: "Retrieve the billing threshold configuration for a project"
---

## Path Parameters

<ParamField path="project_id" type="string" required>
  The unique identifier of the project
</ParamField>

## Response

<ResponseField name="threshold_dollars" type="string">
  The dollar amount threshold that triggers billing
</ResponseField>

<ResponseField name="threshold_days" type="string">
  The number of days threshold that triggers billing
</ResponseField>

<ResponseField name="estimate_days" type="string">
  The estimated number of days before the next billing event
</ResponseField>

<ResponseField name="avg_usage" type="string">
  The average daily usage amount
</ResponseField>

<ResponseField name="usage_days" type="string">
  The number of days since the last billing event
</ResponseField>

<ResponseField name="usage_dollars" type="string">
  The total dollar amount accumulated since the last billing event
</ResponseField>

<ResponseField name="last_billed" type="string">
  The timestamp of the last billing event
</ResponseField>
